GO
CREATE PROCEDURE sp_produto(@codigoOp VARCHAR(1), @codigoTransacao VARCHAR(5), 
				@codigoProduto VARCHAR(5), @quantidade INT, 
				@saida VARCHAR(MAX) OUTPUT)
AS
	DECLARE @tabela VARCHAR(10),
			@query VARCHAR(MAX),
			@erro VARCHAR(MAX),
			@valorProduto INT,
			@valorTotal INT
	
	SET @tabela = 'entrada'
	IF (@codigoOp = 's')
	BEGIN
		SET @tabela = 'saida'
	END
	ELSE
	BEGIN
		IF(@codigoOp = 'e')
		BEGIN
			SET @tabela = 'entrada'
		END
		ELSE
		BEGIN
			SET @tabela = ''
		END
	END
	
	SET @valorTotal = @quantidade * (SELECT valor FROM produto WHERE codigoProduto = @codigoProduto)

	SET @query = 'INSERT INTO '+@tabela+' VALUES ('
			+@codigoTransacao+ ','
			+@codigoProduto+ ','
			+CAST(@quantidade AS VARCHAR(6))+ ','
			+CAST(@valorTotal AS VARCHAR(6))+')'

	PRINT (@query)

	BEGIN TRY
		EXEC (@query)
		SET @saida = 'Transação inserida com sucesso'
	END TRY
	BEGIN CATCH
		RAISERROR('Operaçao Inválida',16,1)
	END CATCH

--- Inserir na tabela entrada
DECLARE @out1 VARCHAR(200)
EXEC sp_produto e, 33, 2, 80, @out1 
PRINT(@out1)


--Inserir na tabela saída
DECLARE @out2 VARCHAR(200)
EXEC sp_produto s, 555, 3, 20, @out2 
PRINT(@out2)

---Operação inválida
DECLARE @out3 VARCHAR(200)
